.data
prompt1: .asciiz " Введите число 1: " # Строка для запроса числа 1 у пользователя
prompt2: .asciiz " Введите число 2: " # Строка для запроса числа 2 у пользователя
result: .asciiz " Результат: " # Строка для вывода результата

.text

Выводим строку prompt1
addi $v0, $zero, 4  # Загружаем в регистр $v0 значение 4 (код запроса вывода строки)
la $a0, prompt1  # Загружаем в регистр $a0 адрес строки prompt1
syscall  # Вызываем системный вызов для печати строки

Считываем число 1
addi $v0, $zero, 5  # Загружаем в регистр $v0 значение 5 (код запроса считывания числа)
syscall  # Вызываем системный вызов для считывания числа
move $s0, $v0  # Сохраняем считанное число в регистре $s0

Выводим строку prompt2
addi $v0, $zero, 4  # Загружаем в регистр $v0 значение 4 (код запроса вывода строки)
la $a0, prompt2  # Загружаем в регистр $a0 адрес строки prompt2
syscall  # Вызываем системный вызов для печати строки

Считываем число 2
addi $v0, $zero, 5  # Загружаем в регистр $v0 значение 5 (код запроса считывания числа)
syscall  # Вызываем системный вызов для считывания числа
move $s1, $v0  # Сохраняем считанное число в регистре $s1

Умножаем числа
multu $s0, $s1  # Умножаем числа из регистров $s0 и $s1

Выводим строку result: 0 - без переполнения, другое значение - с переполнением
addi $v0, $zero, 4  # Загружаем в регистр $v0 значение 4 (код запроса вывода строки)
la $a0, result  # Загружаем в регистр $a0 адрес строки result
syscall  # Вызываем системный вызов для печати строки

addi $v0, $zero, 1  # Загружаем в регистр $v0 значение 1 (код запроса вывода целого числа)
mfhi $a0  # Загружаем в регистр $a0 остаток от деления (результат умножения) из регистра $hi
syscall  # Вызываем системный вызов для печати числа

Завершение программы
li $v0, 10  # Загружаем в регистр $v0 значение 10 (код запроса завершения программы)
syscall  # Вызываем системный вызов для завершения программы
